import { PostList } from '../../components/PostList'
import fs from 'fs'
import path from 'path'
import matter from 'gray-matter'

export const getStaticProps = async () => {
  const postsDirectory = path.join(process.cwd(), 'pages/posts')
  const filenames = fs.readdirSync(postsDirectory)

  const posts = filenames
    .filter(filename => filename.endsWith('.md') || filename.endsWith('.mdx'))
    .filter(filename => filename !== 'index.mdx') // Exclude index page
    .map(filename => {
      const filePath = path.join(postsDirectory, filename)
      const fileContents = fs.readFileSync(filePath, 'utf8')
      const { data } = matter(fileContents)
      
      // Serialize date to string for Next.js getStaticProps
      if (data.date instanceof Date) {
        data.date = data.date.toISOString().split('T')[0]
      }

      return {
        frontMatter: data,
        route: '/posts/' + filename.replace(/\.mdx?$/, '')
      }
    })

  return {
    props: {
      posts
    }
  }
}

<h1 style={{ fontSize: '3rem', fontWeight: 'bold', marginBottom: '2rem', textAlign: 'center' }}>Blog</h1>

<PostList posts={props.posts} />
